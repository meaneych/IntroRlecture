<script>

$('#myslide').on('slideenter', function(){

  $(this).find('article')

    .append('<iframe width="640" height="360" src="http://www.youtube.com/embed/{{{slide.yt}}}?feature=player_detailpage" frameborder="0" allowfullscreen></iframe>')

});

$('#myslide').on('slideleave', function(){

  $(this).find('iframe').remove();

});

</script>
